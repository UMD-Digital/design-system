Loading elements via Javascript rather than as a Web Component

  headline: HTMLElement,
  text: HTMLElement,
  date: HTMLElement,
  actions: HTMLElement,
  isTransparent: Boolean,
  isThemeDark: Boolean,
  isAligned: Boolean,
  hasBorder: Boolean,
});
