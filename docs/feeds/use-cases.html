üè† Use Case 1: Homepage News Section

Display latest news in a featured layout with sidebar.

Implementation
// homepage-news.js
import { news } from '@universityofmaryland/web-feeds-library';
import * as Styles from '@universityofmaryland/web-styles-library';

async function setupHomepageNews() {
  // Create featured news layout
  const newsSection = news.featured({
    token: process.env.UMD_API_TOKEN,
    isLazyLoad: true,
    sidebarItemCount: 5,
    filters: {
      categories: ['featured', 'campus-life'],
      tags: ['homepage']
    }
  });
  
  // Apply layout spacing
  const container = document.querySelector('#homepage-news');
  container.className = 'umd-layout-space-vertical-landing';
  container.appendChild(newsSection.element);
  
  // Handle loading states
  newsSection.element.addEventListener('feed:loading', () => {
    container.insertAdjacentHTML('afterbegin', 
      '
Loading news...
'
    );
  });
  
  newsSection.element.addEventListener('feed:loaded', (e) => {
    container.querySelector('.loading-spinner')?.remove();
    console.log(`Loaded ${e.detail.items.length} news items`);
  });
}

setupHomepageNews();

üìÖ Use Case 2: Department Events Calendar

Create a filterable events display for academic departments.

Implementation
// department-events.js
import { events } from '@universityofmaryland/web-feeds-library';

class DepartmentEventsFeed {
  constructor(containerId) {
    this.container = document.getElementById(containerId);
    this.currentFilter = 'all';
    this.feedInstance = null;
    
    this.init();
  }
  
  init() {
    // Add filter controls
    this.renderFilters();
    
    // Create initial feed
    this.loadEvents();
  }
  
  renderFilters() {
    const filterHTML = `
      <div class="btn-group mb-4" role="group">
        <button class="btn btn-outline-danger active" data-filter="all">
          All Events
        </button>
        <button class="btn btn-outline-danger" data-filter="seminar">
          Seminars
        </button>
        <button class="btn btn-outline-danger" data-filter="workshop">
          Workshops
        </button>
        <button class="btn btn-outline-danger" data-filter="conference">
          Conferences
        </button>
      </div>
      <div id="events-feed"></div>
    `;
    
    this.container.innerHTML = filterHTML;
    
    // Add click handlers
    this.container.querySelectorAll('[data-filter]').forEach(btn => {
      btn.addEventListener('click', (e) => this.handleFilter(e));
    });
  }
  
  handleFilter(e) {
    // Update active state
    this.container.querySelectorAll('.btn').forEach(b => 
      b.classList.remove('active')
    );
    e.target.classList.add('active');
    
    // Reload with filter
    this.currentFilter = e.target.dataset.filter;
    this.loadEvents();
  }
  
  loadEvents() {
    const feedContainer = this.container.querySelector('#events-feed');
    feedContainer.innerHTML = ''; // Clear existing
    
    // Create filtered feed
    this.feedInstance = events.grid({
      token: process.env.UMD_API_TOKEN,
      numberOfColumnsToShow: 3,
      filters: this.currentFilter === 'all' ? {} : {
        eventType: [this.currentFilter]
      },
      sortBy: 'date',
      sortDirection: 'asc',
      showPastEvents: false,
      maxEvents: 9
    });
    
    feedContainer.appendChild(this.feedInstance.element);
  }
}

// Initialize
new DepartmentEventsFeed('department-events');

üìä Use Case 4: Multi-Feed Dashboard

Combine multiple feed types in a unified dashboard.

Implementation
// dashboard.js
import { news, events, academic } from '@universityofmaryland/web-feeds-library';

class FeedDashboard {
  constructor() {
    this.feeds = new Map();
    this.init();
  }
  
  async init() {
    const dashboard = document.getElementById('feed-dashboard');
    
    // Create layout
    dashboard.innerHTML = `
      <div class="row">
        <div class="col-md-8">
          <div class="card mb-4">
            <div class="card-header">
              <h3>Latest News</h3>
            </div>
            <div class="card-body" id="news-feed"></div>
          </div>
        </div>
        <div class="col-md-4">
          <div class="card mb-4">
            <div class="card-header">
              <h3>Upcoming Events</h3>
            </div>
            <div class="card-body" id="events-feed"></div>
          </div>
          <div class="card">
            <div class="card-header">
              <h3>Academic Programs</h3>
            </div>
            <div class="card-body" id="academic-feed"></div>
          </div>
        </div>
      </div>
    `;
    
    // Initialize feeds
    this.loadNewsFeeed();
    this.loadEventsFeed();
    this.loadAcademicFeed();
  }
  
  loadNewsFeeed() {
    const newsFeed = news.list({
      token: process.env.UMD_API_TOKEN,
      maxItems: 5,
      showImages: true,
      showExcerpts: true
    });
    
    this.feeds.set('news', newsFeed);
    document.getElementById('news-feed')
      .appendChild(newsFeed.element);
  }
  
  loadEventsFeed() {
    const eventsFeed = events.list({
      token: process.env.UMD_API_TOKEN,
      maxEvents: 5,
      groupByDate: true,
      showTimes: true
    });
    
    this.feeds.set('events', eventsFeed);
    document.getElementById('events-feed')
      .appendChild(eventsFeed.element);
  }
  
  loadAcademicFeed() {
    const academicFeed = academic.slider({
      token: process.env.UMD_API_TOKEN,
      slidesToShow: 1,
      vertical: true,
      autoPlay: true
    });
    
    this.feeds.set('academic', academicFeed);
    document.getElementById('academic-feed')
      .appendChild(academicFeed.element);
  }
  
  // Refresh all feeds
  refreshAll() {
    this.feeds.forEach(feed => {
      feed.element.dispatchEvent(new Event('feed:refresh'));
    });
  }
}

// Initialize dashboard
const dashboard = new FeedDashboard();

// Refresh feeds every 5 minutes
setInterval(() => dashboard.refreshAll(), 300000);

‚ö†Ô∏è Use Case 5: Robust Error Handling

Handle feed errors gracefully with fallback content.

Implementation
// error-handling.js
import { news } from '@universityofmaryland/web-feeds-library';

class RobustNewsFeed {
  constructor(containerId) {
    this.container = document.getElementById(containerId);
    this.retryCount = 0;
    this.maxRetries = 3;
    
    this.loadFeed();
  }
  
  loadFeed() {
    const feed = news.grid({
      token: process.env.UMD_API_TOKEN,
      numberOfColumnsToShow: 3,
      onError: (error) => this.handleError(error),
      onLoad: (items) => this.handleSuccess(items),
      cacheTimeout: 600000 // 10 minutes
    });
    
    // Show loading state
    this.showLoading();
    
    // Add to container
    this.container.appendChild(feed.element);
  }
  
  showLoading() {
    this.container.innerHTML = `
      <div class="text-center py-5">
        <div class="spinner-border text-danger" role="status">
          <span class="visually-hidden">Loading...</span>
        </div>
        <p class="mt-3">Loading latest news...</p>
      </div>
    `;
  }
  
  handleSuccess(items) {
    this.retryCount = 0;
    
    if (items.length === 0) {
      this.showEmptyState();
    }
  }
  
  handleError(error) {
    console.error('Feed error:', error);
    
    if (this.retryCount < this.maxRetries) {
      this.retryCount++;
      setTimeout(() => this.loadFeed(), 2000 * this.retryCount);
      
      this.container.innerHTML = `
        <div class="alert alert-warning">
          <h5>Loading News...</h5>
          <p>Having trouble connecting. Retrying (${this.retryCount}/${this.maxRetries})...</p>
        </div>
      `;
    } else {
      this.showFallback();
    }
  }
  
  showEmptyState() {
    this.container.innerHTML = `
      <div class="text-center py-5">
        <h4>No News Available</h4>
        <p>Check back later for the latest updates.</p>
        <a href="/news" class="btn btn-outline-danger">
          View News Archive
        </a>
      </div>
    `;
  }
  
  showFallback() {
    this.container.innerHTML = `
      <div class="alert alert-danger">
        <h5>Unable to Load News</h5>
        <p>We're having trouble loading the latest news. Please try again later.</p>
        <div class="mt-3">
          <a href="/news" class="btn btn-danger me-2">View All News</a>
          <button class="btn btn-outline-danger" onclick="location.reload()">
            Retry
          </button>
        </div>
      </div>
    `;
  }
}

// Initialize with error handling
new RobustNewsFeed('news-container');

‚ôæÔ∏è Use Case 6: Infinite Scroll Implementation

Load more content as users scroll.

Implementation
// infinite-scroll.js
import { news } from '@universityofmaryland/web-feeds-library';

const infiniteNews = news.list({
  token: process.env.UMD_API_TOKEN,
  itemsPerPage: 10,
  enableInfiniteScroll: true,
  infiniteScrollThreshold: 200,
  onPageChange: (page) => {
    console.log(`Loading page ${page}`);
    
    // Update URL without reload
    window.history.pushState(
      { page }, 
      '', 
      `?page=${page}`
    );
  }
});

// Add scroll position restoration
window.addEventListener('popstate', (event) => {
  if (event.state?.page) {
    infiniteNews.goToPage(event.state.page);
  }
});
‚úÖ Best Practices
  • Always handle loading and error states
  • Use caching to reduce API calls
  • Implement pagination or infinite scroll for large datasets
  • Test with slow network connections
  • Provide fallback content when feeds fail
  • Use semantic HTML for accessibility
  • Add ARIA live regions for dynamic updates
‚ö° Performance Tips
  • Use lazy loading for below-the-fold feeds
  • Limit initial item counts
  • Enable caching with appropriate timeouts
  • Load critical feeds first
  • Consider server-side rendering for SEO
