üöÄ Use Case 1: Setting Up Base Styles for a New Project

When starting a new UMD project, you need to establish consistent base styles.

Implementation
// styles-setup.js
import * as Styles from '@universityofmaryland/web-styles-library';

export async function initializeStyles() {
  // Load fonts and CSS variables first
  const fonts = Styles.typography.fontFace.base64fonts;
  const preRenderCSS = await Styles.preRenderCss;
  
  // Create and inject base styles
  const baseStyles = document.createElement('style');
  baseStyles.id = 'umd-base-styles';
  baseStyles.textContent = `${fonts} ${preRenderCSS}`;
  document.head.appendChild(baseStyles);
  
  // Load utility classes after DOM is ready
  window.addEventListener('DOMContentLoaded', async () => {
    const postRenderCSS = await Styles.postRenderCss;
    const utilityStyles = document.createElement('style');
    utilityStyles.id = 'umd-utility-styles';
    utilityStyles.textContent = postRenderCSS;
    document.head.appendChild(utilityStyles);
  });
}

// Initialize on import
initializeStyles();

üé® Use Case 2: Building a Theme System

Create light and dark theme variations using UMD design tokens.

Implementation
// theme-manager.js
import { token } from '@universityofmaryland/web-styles-library';

class ThemeManager {
  constructor() {
    this.themes = {
      light: {
        '--theme-bg': '#ffffff',
        '--theme-text': token.color.black,
        '--theme-primary': token.color.red,
        '--theme-secondary': token.color.gold,
        '--theme-border': '#e0e0e0'
      },
      dark: {
        '--theme-bg': '#1a1a1a',
        '--theme-text': '#ffffff',
        '--theme-primary': token.color.red,
        '--theme-secondary': token.color.gold,
        '--theme-border': '#333333'
      }
    };
  }

  apply(themeName) {
    const theme = this.themes[themeName] || this.themes.light;
    const root = document.documentElement;
    
    Object.entries(theme).forEach(([property, value]) => {
      root.style.setProperty(property, value);
    });
    
    root.setAttribute('data-theme', themeName);
  }
  
  toggle() {
    const current = document.documentElement.getAttribute('data-theme');
    this.apply(current === 'dark' ? 'light' : 'dark');
  }
}

export default new ThemeManager();
Usage
// In your component
import themeManager from './theme-manager';

// Apply dark theme
themeManager.apply('dark');

// Toggle between themes
document.querySelector('#theme-toggle').addEventListener('click', () => {
  themeManager.toggle();
});

Use UMD grid utilities and breakpoints for responsive designs.

Implementation
import { token, utilities } from '@universityofmaryland/web-styles-library';

// Generate responsive grid styles
const gridStyles = utilities.create.style.toString({
  '.responsive-container': {
    maxWidth: '1200px',
    margin: '0 auto',
    padding: `0 ${token.spacing.md}`,
    
    [`@media (min-width: ${token.media.breakpoints.tablet})`]: {
      padding: `0 ${token.spacing.lg}`
    },
    
    [`@media (min-width: ${token.media.breakpoints.desktop})`]: {
      padding: `0 ${token.spacing.xl}`
    }
  }
});

üß© Use Case 4: Styling Custom Components

Create consistent component styles using UMD design tokens.

Implementation
// custom-button.js
import { token, animation } from '@universityofmaryland/web-styles-library';

const buttonStyles = {
  '.custom-button': {
    display: 'inline-block',
    padding: `${token.spacing.sm} ${token.spacing.lg}`,
    backgroundColor: token.color.red,
    color: '#ffffff',
    fontFamily: token.font.family.sans,
    fontSize: token.font.size.base,
    fontWeight: token.font.weight.bold,
    textDecoration: 'none',
    border: 'none',
    borderRadius: '4px',
    cursor: 'pointer',
    transition: 'all 0.3s ease',
    
    '&:hover': {
      backgroundColor: token.color.redDark,
      transform: 'translateY(-2px)',
      boxShadow: '0 4px 8px rgba(0, 0, 0, 0.2)'
    },
    
    '&:active': {
      transform: 'translateY(0)',
      boxShadow: '0 2px 4px rgba(0, 0, 0, 0.2)'
    },
    
    '&.secondary': {
      backgroundColor: 'transparent',
      color: token.color.red,
      border: `2px solid ${token.color.red}`,
      
      '&:hover': {
        backgroundColor: token.color.red,
        color: '#ffffff'
      }
    }
  }
};

// Convert to CSS
const css = await utilities.create.style.toString(buttonStyles);

üé® Use Case 7: Working with Colors

Access and use UMD brand colors effectively.

Color Tokens

Red: #e21833

Gold: #FFD200

Black: #000000

White: #ffffff

Usage Examples
import { token } from '@universityofmaryland/web-styles-library';

// In JavaScript
const styles = {
  backgroundColor: token.color.red,
  color: token.color.white,
  borderColor: token.color.gold
};

// As CSS variables
:root {
  --primary: var(--color-red);
  --secondary: var(--color-gold);
  --text: var(--color-black);
}
‚úÖ Best Practices
  • Always use design tokens instead of hard-coded values
  • Load fonts and variables before other styles
  • Use utility classes when possible before creating custom styles
  • Test responsive designs at all UMD breakpoints
  • Ensure color combinations meet WCAG contrast requirements
